# Extract merkle root into text file 

jq -r .root merkle_data.json > root.txt

# TPM signature time 

/usr/bin/time -f "wall=%E  user=%U  sys=%S" \
  tpm2_sign -c ../aik_loaded.ctx -g sha256 \
    -o ../root.sig root.txt

#TPM Verification with time 

/usr/bin/time -f "Verification time: %E User Time: %U Sys Time: %S" \
tpm2_verifysignature -c ../aik_loaded.ctx -g sha256 -s ../root.sig -m root.txt



# Ask the TPM to sign it with your AIK 

tpm2_sign \
  --key-context aik_loaded.ctx \
  --hash-algorithm sha256 \
  --signature root.sig \
  root.txt


# Verify TPM's signature

tpm2_verifysignature \
  --public aik_pub.pem \
  --hash-algorithm sha256 \
  --signature root.sig \
  --message root.txt




# Provision an attestation key in TPM


# 4.1 Create a TPM “primary” storage key (under the owner hierarchy)
tpm2_createprimary -C o -c primary.ctx

# 4.2 Create an AIK (Attestation Identity Key) under that primary key
tpm2_create -C primary.ctx \
  -u aik.pub -r aik.priv \
  -c aik.ctx \
  -G rsa

# 4.3 Load the AIK into the TPM
tpm2_load -C primary.ctx \
  -u aik.pub -r aik.priv \
  -c aik_loaded.ctx

# 4.4 Extract the public key for later verification
tpm2_readpublic -c aik_loaded.ctx \
  -f pem -o aik_pub.pem
